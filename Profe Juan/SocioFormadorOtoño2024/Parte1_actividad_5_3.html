sesiones = [sesin1, sesin2, sesin3, sesin4]
distancias_totales = []

#Calcular la distancia total para cada sesin
for session in sesiones:
    #Calcular distancia entre puntos consecutivos en el espacio 3D
    dif_x = session['Des_x'].diff().fillna(0)
    dif_y = session['Des_y'].diff().fillna(0)
    dif_z = session['Des_z'].diff().fillna(0)
    distancia = np.sqrt(dif_x**2 + dif_y**2 + dif_z**2).sum()  # Suma total de distancias
    distancias_totales.append(distancia)

#Configuracin de la figura 2x2
fig, axs = plt.subplots(nrows=2, ncols=2, figsize=(20, 15))
plt.style.use('dark_background')
colors = plt.get_cmap("twilight").colors

#Grfica de distancia total por cada sesin
for i in range(4):

    row = i // 2
    col = i % 2
    ax = axs[row][col]

    #Grfico de barras de la distancia total de la sesin
    bars = ax.bar([f"Sesin {i + 1}"], [distancias_totales[i]], color=colors[i * 50])  # Elegimos un color por sesin

    #Configuracin del grfico
    ax.set_ylim(0, max(distancias_totales) * 1.1)  # Margen sobre el mximo valor
    ax.set_title(f'Distancia Total - Sesin {i + 1}', fontsize=14)
    ax.set_ylabel("Distancia Total Recorrida")

    #Mostrar el valor encima de la barra
    for bar in bars:
        ax.text(
            bar.get_x() + bar.get_width() / 2,
            bar.get_height(),
            f'{bar.get_height():.1f}',
            ha='center', va='bottom'
        )

#Ajuste del espacio entre subplots
plt.tight_layout()
plt.show()

sesiones = [sesin1, sesin2, sesin3, sesin4]
emociones = ['01_C', '02_A', '03_D', '04_M']
tiempos_emociones_totales = {em: [] for em in emociones}

#calcular el tiempo total de cada emocin en cada sesin
for session in sesiones:
    for em in emociones:
        # calcular tiempo acumulado para esta emocin en la sesin actual
        tiempo_total = 0
        start = None
        for i in range(len(session)):
            if session[em].iloc[i] == 1:
                if start is None:
                    start = session['time'].iloc[i]
            else:
                if start is not None:
                    end = session['time'].iloc[i - 1]
                    tiempo_total += end - start
                    start = None
        #capturar el ltimo intervalo si termina con un "1" en la ltima fila
        if start is not None:
            tiempo_total += session['time'].iloc[-1] - start

        #guardar el tiempo total de la emocin en la sesin
        tiempos_emociones_totales[em].append(tiempo_total)

#creo grfica comparativa para cada emocin en cada sesin
fig, axs = plt.subplots(nrows=2, ncols=2, figsize=(20, 15))
plt.style.use('dark_background')
colors = plt.get_cmap("twilight").colors

for i, em in enumerate(emociones):
    row = i // 2
    col = i % 2
    ax = axs[row][col]

    #Graficar tiempo total de la emocin en cada sesin
    bars = ax.bar([f"Sesin {j+1}" for j in range(len(sesiones))], tiempos_emociones_totales[em], color=colors[i * 50])

    #Configuracin de la grfica
    ax.set_title(f'Tiempo Total en Emocin {em}', fontsize=14)
    ax.set_xlabel("Sesin")

    #Mostrar el tiempo encima de cada barra

#creo la figura con 4 subplots en una columna
fig, axs = plt.subplots(nrows=4, ncols=1, figsize=(10, 12))

#itero sobre cada sesin y su subplot
for s, ax in enumerate(axs):
    #sesin actual
    session_data = sesiones[s]

    #grafico cada columna de sentimiento en el subplot correspondiente
    for i, col in enumerate(sentiment_columns):
        #filtro los intervalos de tiempo donde el sentimiento est presente (valor = 1)
        intervalos = []
        start = None
        for j in range(len(session_data)):
            if session_data[col].iloc[j] == 1:
                if start is None:
                    start = session_data['time'].iloc[j]
            else:
                if start is not None:
                    end = session_data['time'].iloc[j - 1]
                    intervalos.append((start, end))
                    start = None
        #capturo el ltimo intervalo si termina en el ltimo valor de la columna
        if start is not None:
            intervalos.append((start, session_data['time'].iloc[-1]))

        #dibujar barras horizontales para los intervalos
        for (start, end) in intervalos:
            ax.hlines(y=i, xmin=start, xmax=end, color=plt.get_cmap("Set1")(i), linewidth=6)

    #Etiquetas y formato
    ax.set_yticks(range(len(sentiment_columns)))
    ax.set_yticklabels(sentiment_columns)
    ax.set_xlabel("Tiempo")
    ax.set_title(f"Presencia de sentimientos en intervalos de tiempo - Sesin {s + 1}")
    ax.grid(axis='x', linestyle='--', alpha=0.7)


plt.tight_layout()

sesiones = [sesin1, sesin2, sesin3, sesin4]
emociones = ['01_C', '02_A', '03_D', '04_M']
duraciones_emociones = {em: [[] for _ in range(4)] for em in emociones}

#calcular la duracin de cada intervalo para cada emocin en cada sesin
for s, session in enumerate(sesiones):
    for em in emociones:
        duraciones = []
        start = None
        for i in range(len(session)):
            if session[em].iloc[i] == 1:
                if start is None:
                    start = session['time'].iloc[i]
            else:
                if start is not None:
                    end = session['time'].iloc[i - 1]
                    duraciones.append(end - start)
                    start = None
        #Capturar el ltimo intervalo si termina con un "1" en la ltima fila
        if start is not None:
            duraciones.append(session['time'].iloc[-1] - start)

        #Guardar las duraciones de la emocin en la sesin
        duraciones_emociones[em][s] = duraciones

#crear los histogramas de distribucin de tiempos para cada emocin y sesin
fig, axs = plt.subplots(nrows=2, ncols=2, figsize=(20, 15))
plt.style.use('dark_background')
colors = plt.get_cmap("twilight").colors

for i, em in enumerate(emociones):
    row = i // 2
    col = i % 2
    ax = axs[row][col]

    #Histograma de cada emocin en todas las sesiones
    for s in range(4):
        ax.hist(duraciones_emociones[em][s], bins=10, alpha=0.6, label=f"Sesin {s + 1}", color=colors[s * 50])

    #Configuracin del grfico
    ax.set_title(f'Distribucin de Tiempos de Emocin {em}', fontsize=14)
    ax.set_xlabel("Duracin de Intervalos (segundos)")
    ax.set_ylabel("Frecuencia")
    ax.legend()

#Ajustar el espacio entre subplots
plt.tight_layout()
plt.show()
