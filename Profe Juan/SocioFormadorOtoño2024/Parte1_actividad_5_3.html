sesiones = [sesión1, sesión2, sesión3, sesión4]
distancias_totales = []

#Calcular la distancia total para cada sesión
for session in sesiones:
    #Calcular distancia entre puntos consecutivos en el espacio 3D
    dif_x = session['Des_x'].diff().fillna(0)
    dif_y = session['Des_y'].diff().fillna(0)
    dif_z = session['Des_z'].diff().fillna(0)
    distancia = np.sqrt(dif_x**2 + dif_y**2 + dif_z**2).sum()  # Suma total de distancias
    distancias_totales.append(distancia)

#Configuración de la figura 2x2
fig, axs = plt.subplots(nrows=2, ncols=2, figsize=(20, 15))
plt.style.use('dark_background')
colors = plt.get_cmap("twilight").colors

#Gráfica de distancia total por cada sesión
for i in range(4):

    row = i // 2
    col = i % 2
    ax = axs[row][col]

    #Gráfico de barras de la distancia total de la sesión
    bars = ax.bar([f"Sesión {i + 1}"], [distancias_totales[i]], color=colors[i * 50])  # Elegimos un color por sesión

    #Configuración del gráfico
    ax.set_ylim(0, max(distancias_totales) * 1.1)  # Margen sobre el máximo valor
    ax.set_title(f'Distancia Total - Sesión {i + 1}', fontsize=14)
    ax.set_ylabel("Distancia Total Recorrida")

    #Mostrar el valor encima de la barra
    for bar in bars:
        ax.text(
            bar.get_x() + bar.get_width() / 2,
            bar.get_height(),
            f'{bar.get_height():.1f}',
            ha='center', va='bottom'
        )

#Ajuste del espacio entre subplots
plt.tight_layout()
plt.show()

sesiones = [sesión1, sesión2, sesión3, sesión4]
emociones = ['01_C', '02_A', '03_D', '04_M']
tiempos_emociones_totales = {em: [] for em in emociones}

#calcular el tiempo total de cada emoción en cada sesión
for session in sesiones:
    for em in emociones:
        # calcular tiempo acumulado para esta emoción en la sesión actual
        tiempo_total = 0
        start = None
        for i in range(len(session)):
            if session[em].iloc[i] == 1:
                if start is None:
                    start = session['time'].iloc[i]
            else:
                if start is not None:
                    end = session['time'].iloc[i - 1]
                    tiempo_total += end - start
                    start = None
        #capturar el último intervalo si termina con un "1" en la última fila
        if start is not None:
            tiempo_total += session['time'].iloc[-1] - start

        #guardar el tiempo total de la emoción en la sesión
        tiempos_emociones_totales[em].append(tiempo_total)

#creo gráfica comparativa para cada emoción en cada sesión
fig, axs = plt.subplots(nrows=2, ncols=2, figsize=(20, 15))
plt.style.use('dark_background')
colors = plt.get_cmap("twilight").colors

for i, em in enumerate(emociones):
    row = i // 2
    col = i % 2
    ax = axs[row][col]

    #Graficar tiempo total de la emoción en cada sesión
    bars = ax.bar([f"Sesión {j+1}" for j in range(len(sesiones))], tiempos_emociones_totales[em], color=colors[i * 50])

    #Configuración de la gráfica
    ax.set_title(f'Tiempo Total en Emoción {em}', fontsize=14)
    ax.set_xlabel("Sesión")
    ax.set_ylabel("Tiempo Total (segundos)")

    #Mostrar el tiempo encima de cada barra
    for bar in bars:
        ax.text(
            bar.get_x() + bar.get_width() / 2,
            bar.get_height(),
            f'{bar.get_height():.1f}',
            ha='center', va='bottom'
        )

plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt

sesiones = [sesión1, sesión2, sesión3, sesión4]
sentiment_columns = ['01_C', '02_A', '03_D', '04_M']

#creo la figura con 4 subplots en una columna
fig, axs = plt.subplots(nrows=4, ncols=1, figsize=(10, 12))

#itero sobre cada sesión y su subplot
for s, ax in enumerate(axs):
    #sesión actual
    session_data = sesiones[s]

    #grafico cada columna de sentimiento en el subplot correspondiente
    for i, col in enumerate(sentiment_columns):
        #filtro los intervalos de tiempo donde el sentimiento está presente (valor = 1)
        intervalos = []
        start = None
        for j in range(len(session_data)):
            if session_data[col].iloc[j] == 1:
                if start is None:
                    start = session_data['time'].iloc[j]
            else:
                if start is not None:
                    end = session_data['time'].iloc[j - 1]
                    intervalos.append((start, end))
                    start = None
        #capturo el último intervalo si termina en el último valor de la columna
        if start is not None:
            intervalos.append((start, session_data['time'].iloc[-1]))

        #dibujar barras horizontales para los intervalos
        for (start, end) in intervalos:
            ax.hlines(y=i, xmin=start, xmax=end, color=plt.get_cmap("Set1")(i), linewidth=6)

    #Etiquetas y formato
    ax.set_yticks(range(len(sentiment_columns)))
    ax.set_yticklabels(sentiment_columns)
    ax.set_xlabel("Tiempo")
    ax.set_title(f"Presencia de sentimientos en intervalos de tiempo - Sesión {s + 1}")
    ax.grid(axis='x', linestyle='--', alpha=0.7)


plt.tight_layout()
plt.show()

sesiones = [sesión1, sesión2, sesión3, sesión4]
emociones = ['01_C', '02_A', '03_D', '04_M']
duraciones_emociones = {em: [[] for _ in range(4)] for em in emociones}

#calcular la duración de cada intervalo para cada emoción en cada sesión
for s, session in enumerate(sesiones):
    for em in emociones:
        duraciones = []
        start = None
        for i in range(len(session)):
            if session[em].iloc[i] == 1:
                if start is None:
                    start = session['time'].iloc[i]
            else:
                if start is not None:
                    end = session['time'].iloc[i - 1]
                    duraciones.append(end - start)
                    start = None
        #Capturar el último intervalo si termina con un "1" en la última fila
        if start is not None:
            duraciones.append(session['time'].iloc[-1] - start)

        #Guardar las duraciones de la emoción en la sesión
        duraciones_emociones[em][s] = duraciones

#crear los histogramas de distribución de tiempos para cada emoción y sesión
fig, axs = plt.subplots(nrows=2, ncols=2, figsize=(20, 15))
plt.style.use('dark_background')
colors = plt.get_cmap("twilight").colors

for i, em in enumerate(emociones):
    row = i // 2
    col = i % 2
    ax = axs[row][col]

    #Histograma de cada emoción en todas las sesiones
    for s in range(4):
        ax.hist(duraciones_emociones[em][s], bins=10, alpha=0.6, label=f"Sesión {s + 1}", color=colors[s * 50])

    #Configuración del gráfico
    ax.set_title(f'Distribución de Tiempos de Emoción {em}', fontsize=14)
    ax.set_xlabel("Duración de Intervalos (segundos)")
    ax.set_ylabel("Frecuencia")
    ax.legend()

#Ajustar el espacio entre subplots
plt.tight_layout()
plt.show()
